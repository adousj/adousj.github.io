lines = open('a.txt').readlines()

result = ''
for line in lines :
    line = line[:-1].split(' ')
    for w in line:
        ch = 0
        k = 0
        print len(w),
        for i in w :
            if i =='0' :
                i = 0
            elif i == '\n':
                break
            else :
                i = 1
            ch = ch*2 + i
            k += 1
        result = result + str(chr(ch))

open('out.txt','w').write(result)
