import threading
import time

class GetKey(threading.Thread):
    ''''''
    def __init__(self):
        threading.Thread.__init__(self)
        self.lock = threading.Lock()
        self.setDaemon(True)

    def run(self):
        global strin
        strin = raw_input()

class WaitKey(threading.Thread):
    ''''''
    def __init__(self,t):
        self.t = t
        threading.Thread.__init__(self)

    def run(self):
        GetKey().start()
        time.sleep(self.t)

def waitKey(t):
    global strin
    strin = ""
    thrd = WaitKey(t/1000.0)
    thrd.start()
    thrd.join()
    return strin

if __name__ == "__main__":
    print waitKey(3000)
