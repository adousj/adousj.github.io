# !/usr/bin/ruby

class Greeter
    def initialize(name)
        @name = name
    end

    def say_hi
        puts "hello,#{@name}"
    end

    def say_bye
        puts "bye, #{@name}, see you soon."
    end
end

g = Greeter.new("adou")
g.say_hi()
g.say_bye()
